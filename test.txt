"use client";

import { motion } from "framer-motion";
import { useEffect, useState } from "react";
import Image from "next/image";

export default function HeroSection() {
  const [cursor, setCursor] = useState({ x: 50, y: 50 });

  // Track mouse position for spotlight
  useEffect(() => {
    const handleMove = (e: MouseEvent) => {
      setCursor({
        x: (e.clientX / window.innerWidth) * 100,
        y: (e.clientY / window.innerHeight) * 100,
      });
    };
    window.addEventListener("mousemove", handleMove);
    return () => window.removeEventListener("mousemove", handleMove);
  }, []);

  return (
    <section className="relative h-screen flex items-center justify-center text-white overflow-hidden">
      {/* ðŸŽ¨ Animated Gradient Glow */}
      <motion.div
        className="absolute inset-0"
        style={{
          background:
            "linear-gradient(135deg, #1DB954 0%, #191414 50%, #3D3D3D 100%)",
          backgroundSize: "200% 200%",
        }}
        animate={{ backgroundPosition: ["0% 50%", "100% 50%", "0% 50%"] }}
        transition={{ duration: 15, repeat: Infinity, ease: "linear" }}
      />

      {/* ðŸŽ¶ Pulsing Glow Overlay */}
      <motion.div
        className="absolute inset-0 bg-gradient-to-r from-green-500/20 via-purple-500/20 to-black/70"
        animate={{ opacity: [0.6, 0.9, 0.6] }}
        transition={{ duration: 2, repeat: Infinity, ease: "easeInOut" }}
      />

      {/* ðŸ“Š Equalizer Bars */}
      <div className="absolute bottom-0 flex w-full h-40 items-end justify-center gap-1">
        {Array.from({ length: 30 }).map((_, i) => (
          <motion.div
            key={i}
            className="w-1 bg-green-400/40 rounded-t"
            animate={{
              height: [10, 80, 30, 60, 20],
            }}
            transition={{
              duration: 1 + Math.random(),
              repeat: Infinity,
              ease: "easeInOut",
              delay: i * 0.1,
            }}
          />
        ))}
      </div>

      {/* âœ¨ Cursor Spotlight */}
      <div
        className="absolute inset-0 pointer-events-none"
        style={{
          background: `radial-gradient(circle at ${cursor.x}% ${cursor.y}%, rgba(29,185,84,0.25), transparent 40%)`,
        }}
      />

      {/* Dark overlay for readability */}
      <div className="absolute inset-0 bg-black/40" />

      {/* ðŸ”¥ Hero Content */}
      <div className="relative z-10 text-center px-6">
        {/* Example "Now Playing" card */}
        <motion.div
          className="bg-neutral-900/80 backdrop-blur-md rounded-2xl shadow-lg p-6 inline-flex items-center gap-4"
          initial={{ opacity: 0, y: 30 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 1 }}
        >
          <Image
            src="/assets/album-art.jpg"
            alt="Album Art"
            width={80}
            height={80}
            className="rounded-md"
          />
          <div className="text-left">
            <h3 className="text-lg font-semibold">Now Playing</h3>
            <p className="text-sm text-neutral-400">Artist Name</p>
          </div>
        </motion.div>

        <h1 className="mt-10 text-5xl md:text-6xl font-bold">
          Visuals that <span className="text-green-500">Resonate</span>
        </h1>
        <p className="mt-4 text-lg text-neutral-300 max-w-xl mx-auto">
          A Spotify-inspired portfolio showcasing films, visual art, and creative
          storytelling.
        </p>
      </div>
    </section>
  );
}
